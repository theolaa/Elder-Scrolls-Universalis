estate_cossacks_disaster = {
	potential = {
		has_estate = estate_cossacks
		OR = {
			NOT = { has_country_flag = cossack_estate_triggered }
			had_country_flag = {
				flag = cossack_estate_triggered
				days = 365
			}
		}
	}


	can_start = {
		has_any_disaster = no
		estate_influence = {
			estate = estate_cossacks
			influence = 80
		}
	}
	
	can_stop = {
		OR = {
			has_any_disaster = yes
			NOT = {
				estate_influence = {
					estate = estate_cossacks
					influence = 50
				}		
			}
		}
	}
	
	down_progress = {
		factor = 1
	}
	
	progress = {
		modifier = {
			factor = 5
			estate_influence = {
				estate = estate_cossacks
				influence = 50
			}
		}		
	}
	
	can_end = {
		always = yes
	}

	on_start = estate_disasters.10
	
	on_start_effect = {
		if = {
			limit = {
				has_dlc = "The Cossacks"
			}
			custom_tooltip = estate_cossacks_disaster_tooltip_cossacks
		}
		else = {
			custom_tooltip = estate_cossacks_disaster_tooltip
		}
	}
	
	on_monthly = {
	}
}
