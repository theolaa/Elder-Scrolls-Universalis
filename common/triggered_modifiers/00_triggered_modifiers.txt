# Discovery of the Akaviri trade route.
#######################################################################

east_akaviri_trade_route = {
	potential = {
		num_of_ports = 1
		NOT = {
			capital_scope = {
				continent = Akavir
			}
		}
	}

	trigger = {
		has_discovered = 702
		num_of_ports = 5
		has_idea = quest_for_the_new_world
		any_active_trade_node = {
			has_merchant = ROOT
			OR = {
				province_id = 702
				province_id = 611
				province_id = 743
				province_id = 583
				province_id = 468
			}
		}
	}

	trade_efficiency = 0.05	
}

# Crusader
#######################################################################

crusader = {
	potential = {
		religion = chimer_pantheon
		crusade_target = {
			exists = yes
		}
		
	}

	trigger = {
		
		crusade_target = {
			war_with = ROOT
			any_province = {
				controlled_by = root
			}
		}  
		land_maintenance = 1.0
	}

	global_manpower_modifier = 0.35
	global_tax_modifier = 0.10
	land_morale = 0.10
	prestige = 1
	papal_influence = 1
	devotion = 0.25
	#icon = 9
}

		}

		750 = { religion = dragon_cult }

		1288 = { religion = dragon_cult}
		724 = { religion = dragon_cult }
	}

}
