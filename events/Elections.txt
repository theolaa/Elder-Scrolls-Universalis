		change_dip = 1
		change_mil = 1

			limit = {
				is_tribal = no  
				government = republic

			limit = {
				is_tribal = no 
				government = republic

				add_dip_power = 50

				add_mil_power = 50
			}
		}

	}

				mil = 1
				adm = 4
				dip = 1
				fixed = yes
				random_gender = yes # Allows random gender for republics
			}
		}
		else = {
			define_ruler = {
				mil = 1
				adm = 4
				dip = 1					
				fixed = yes
			}
		}
		if = {
			limit = { 
				has_faction = mr_guilds 
			}
			add_faction_influence = {
				faction = mr_guilds
				influence = 20
			}				
		}
		add_karma = 10
	}
	option = {
		name = "EVTOPTC700"	# A diplomat
		ai_chance = { factor = 20 }
		if = {
			limit = { has_active_policy = mandatory_service }
			define_ruler = {
				dip = 4
				adm = 1
				mil = 1					
				fixed = yes
				random_gender = yes # Allows random gender for republics
			}
		}
		else = {
			define_ruler = {
				dip = 4
				adm = 1
				mil = 1					
				fixed = yes
			}
		}
		if = {
			limit = { 
				has_faction = mr_traders 
			}
			add_faction_influence = {
				faction = mr_traders
				influence = 20
			}				
		}
		add_karma = 10		
	}
	option = {
		name = "EVTOPTD700"	# A military man
		ai_chance = { factor = 20 }
		if = {

			define_ruler = {

# Lottery Election by Daniel Tolman
country_event = {
	id = elections.800
	title = elections.800.t
	desc = elections.800.d
	picture = ELECTION_REPUBLICAN_eventPicture
	
	is_triggered_only = yes
	
	trigger = {
		has_government_attribute = lottery_election
	}
	
	immediate = {
		hidden_effect = {
			if = {
				limit = {
					has_active_policy = mandatory_service
				}
				define_exiled_ruler = {
					exiled_as = lottery_candidate_a_@ROOT

					option = 0
				}

				limit = {
					has_active_policy = mandatory_service
				}

					exiled_as = lottery_candidate_b_@ROOT

	}
	
	after = {
		clear_exiled_ruler = lottery_candidate_a_@ROOT

			factor = 0
		}
