	    tag = RED
	    NOT = { has_country_modifier = sand_storm_1 }
	}
	
	mean_time_to_happen = {
		months = 1200
	}
	
	option = {
		name = "es_terrain.1.a"
		ai_chance = { factor = 100 }
		add_country_modifier = {
				name = sand_storm_1
				duration = 36500
		    }

		    limit = {
		        region = po_tun_lr
			}

				name = sand_storm
				duration = 9125
			}
	    }
		every_country = {
			limit = {

				    region = po_tun_lr
				}
			}
