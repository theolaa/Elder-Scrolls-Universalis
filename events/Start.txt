country_event = {
	id = es_start.1
	title = es_start.1.t
	desc = es_start.1.d
	picture = ES_START_eventPicture
	
	fire_only_once = yes

	trigger = {
	ai = no
	}
	
	mean_time_to_happen = {
		months = 1
	}
	
	immediate = {
        if = {
		limit = {
		NOT = { is_year = 50 }
		}
		add_country_modifier = {
		name = "antibankrapucy"
		duration = 1460
		}
		}	
		if = {
		limit = {
		NOT = { is_year = 50 }
		capital_scope = { superregion = pyandonea_superregion }
		}
		pyandonea_superregion = {
		    discover_country = ROOT
		}
		}
		if = {
		limit = {
		NOT = { is_year = 50 }
		capital_scope = { superregion = tamriel_superregion }
		}
		tamriel_superregion = {
		    discover_country = ROOT
		}
		}
		if = {
		limit = {
		NOT = { is_year = 50 }
		capital_scope = { superregion = yokuda_superregion }
		}
		yokuda_superregion = {
		    discover_country = ROOT
		}
		}
		if = {
		limit = {
		NOT = { is_year = 50 }
		capital_scope = { superregion = atmora_superregion }
		}
		atmora_superregion = {
		    discover_country = ROOT
		}
		}
		if = {
		limit = {
		NOT = { is_year = 50 }
		capital_scope = { superregion = akavir_superregion }
		}
		akavir_superregion = {
		    discover_country = ROOT
		}
		}
	}
	
	
	option = {
		name = es_start.1.a	
	}
}

country_event = {
	id = es_start.2
	title = es_start.1.t
	desc = es_start.1.d
	picture = ES_START_eventPicture
	
	fire_only_once = yes

	trigger = {
	tag = RED
	}
	
	mean_time_to_happen = {
		months = 1
	}
	
	immediate = {
	    hidden_effect = {
		add_country_modifier = {
				name = "red_mountain_eruption"
				duration = 36500
				hidden = yes
		}	
		add_country_modifier = {
				name = "rise_time"
				duration = 365
				hidden = yes
			}
        add_country_modifier = {
				name = "crisis_time"
				duration = 365
				hidden = yes
			}	
        add_country_modifier = {
				name = "trade_time"
				duration = 365
				hidden = yes
			}
        every_province = {
		        limit = {
		        OR = {
				area = morrowind_21_la
				area = morrowind_20_la
				area = morrowind_19_la
				area = morrowind_18_la
				}
			    }
			add_province_modifier = {
					name = "ever-ashland"
					duration = -1
			    }
		}			
        }
	}
	
	
	option = {
		name = es_start.1.a	
	}
}
