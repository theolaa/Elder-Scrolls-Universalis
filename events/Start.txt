namespace = es_start

country_event = {
	id = es_start.1
	title = es_start.1.t
	desc = es_start.1.d
	picture = ES_START_eventPicture
	

	trigger = {
	ai = no
	NOT = { has_country_flag = es_start_ai_no_flag }
	}
	
	mean_time_to_happen = {
		months = 1
	}
	
	immediate = {
	    hidden_effect = {
        if = {
		limit = {
		is_bankrupt = yes
		}
		add_country_modifier = {
		name = "antibankrapucy"
		duration = 1460
		}
		}	
		if = {
		limit = {
		NOT = { is_year = 50 }
		capital_scope = { superregion = pyandonea_superregion }
		}
		pyandonea_superregion = {
		    discover_country = ROOT
		}
		}
		if = {
		limit = {
		NOT = { is_year = 50 }
		capital_scope = { superregion = tamriel_superregion }
		}
		tamriel_superregion = {
		    discover_country = ROOT
		}
		}
		if = {
		limit = {
		NOT = { is_year = 50 }
		capital_scope = { superregion = yokuda_superregion }
		}
		yokuda_superregion = {
		    discover_country = ROOT
		}
		}
		if = {
		limit = {
		NOT = { is_year = 50 }
		capital_scope = { superregion = atmora_superregion }
		}
		atmora_superregion = {
		    discover_country = ROOT
		}
		}
		if = {
		limit = {
		NOT = { is_year = 50 }
		capital_scope = { superregion = akavir_superregion }
		}
		akavir_superregion = {
		    discover_country = ROOT
		}
		}
		
		if = {
		limit = {
		is_year = 266
		}
		set_global_flag = ayleid_fall_flag
		}
		if = {
		limit = {
		is_year = 369
		}
		set_global_flag = marukhism_flag
		}
		if = {
		limit = {
		is_year = 242
		}
		set_global_flag = etherial_crisis_flag
		set_global_flag = alessian_rebellion_flag
		}
		if = {
		limit = {
		is_year = 609
		}
		set_global_flag = fall_of_direnii_flag
		}
		if = {
		limit = {
		is_year = 750
		}
		set_global_flag = dwemer_dissapearence_flag
		}
		
		if = {
		limit = {
		is_free_or_tributary_trigger = yes
		}
		set_country_flag = arrow_in_the_knee_flag
		}
		
		every_owned_province = {
		remove_estate = estate_church
		remove_estate = estate_nobles
		remove_estate = estate_burghers
		remove_estate = estate_cossacks
		remove_estate = estate_nomadic_tribes
		}
		
		set_country_flag = es_start_ai_no_flag
		
		add_country_modifier = {
			name = "vassal_autonomy"
			duration = 9125
		}
		
		add_country_modifier = {
			name = "es_female_advisors"
			duration = -1
			hidden = yes
		}
		}
	}
	
	
	option = {
		name = es_start.1.a	
	}
}
